            // Scroll-driven dot activation + progress bar
            function onScroll() {
                if (!roadway) return;
                var rect = roadway.getBoundingClientRect();
                var roadwayTop = rect.top;
                var roadwayH = rect.height;
                var viewH = window.innerHeight;
                var scrolled = (viewH * 0.5 - roadwayTop) / roadwayH;
                scrolled = Math.max(0, Math.min(1, scrolled));
                if (progress) progress.style.height = (scrolled * 100) + '%';

                dots.forEach(function (dot, i) {
                    var stopRect = stops[i].getBoundingClientRect();
                    var stopCenter = stopRect.top + stopRect.height / 2;
                    if (stopCenter < viewH * 0.65) {
                        dot.classList.add('active');
                    } else {
                        dot.classList.remove('active');
                    }
                });
            }
            window.addEventListener('scroll', onScroll, { passive: true });
            onScroll();

            // Open popup
            function openModal(index) {
                var d = cardData[index];
                if (!d) return;
                var html = '';
                html += '<h4>' + d.title + '</h4>';
                html += '<p class="modal-desc">' + d.desc + '</p>';
                html += '<img class="modal-img" src="' + d.img + '" alt="' + d.title + '" />';
                html += '<ul>';
                d.bullets.forEach(function (b) { html += '<li>' + b + '</li>'; });
                html += '</ul>';
                modalBody.innerHTML = html;
                overlay.classList.add('open');
                document.body.style.overflow = 'hidden';
            }

            // Close popup
            function closeModal() {
                overlay.classList.remove('open');
                document.body.style.overflow = '';
            }

            // Attach click to each stop
            stops.forEach(function (stop) {
                stop.addEventListener('click', function () {
                    var idx = parseInt(stop.getAttribute('data-card'), 10);
                    openModal(idx);
                });
            });

            // Close on X button
            if (closeBtn) closeBtn.addEventListener('click', closeModal);

            // Close on overlay click (anywhere outside modal)
            if (overlay) overlay.addEventListener('click', function (e) {
                if (e.target === overlay) closeModal();
            });

            // Close on Escape key
            document.addEventListener('keydown', function (e) {
                if (e.key === 'Escape') closeModal();
            });
        })();
